#include "UTAddOn/scripts/ut_helper.txt"

namespace UTAddOn
{

enum UT_Difficulty
{
    UTA_DIFF_OFF = 0,
    UTA_DIFF_BABY = 10,
    UTA_DIFF_EASY = 25,
    UTA_DIFF_NORMAL = 50,
    UTA_DIFF_HARD = 75,
    UTA_DIFF_HARDCORE = 100
}


// Dynamic gameplay settings
class UT_GamePlaySettings
{
    bool enableDamage = true;
    bool enableAchievements = true;
    bool enableHealing = true;
    bool enableArmorHealing = true;
    bool enableHud = true;
    bool enableDamageInterrupt = true;
    float streakDuration = 2.5;
    float maxHealRate = 1.0f / 10.0f;
    int healthDmg_60 = 1;   // dt = 1.0
    int healthDmg_Norm = 3; // dt = 2.0
    int healthDmg_Over = 5; // dt = 2.0
    int armorDmg_Norm = 2;  // dt = 2.0
    int armorDmg_Over = 2;  // dt = 1.0
}
UT_GamePlaySettings g_gamePlaySettings; // Global variable for settings storage


// Constant parameters, not intended to be changed during game
class UT_GamePlayParameters
{
    int par_initialMaxHealth = 100;
    int par_initialMaxArmor = 5;
    int par_upperBoundHealth = 250;
    int par_upperBoundArmor = 50;
}
const UT_GamePlayParameters g_gamePlayParams;   // Global (const) variable for parameter access


// Data, which is needed to be stored (and loaded) in save files
class UT_SaveData
{

    // =========== PROPERTIES GET/SET ===========

    // Max heal health
    // ---------------
    int maxHealHealth
    {
        get const 
        {
            int val;
            GetGV("UTAddOn_save_maxHealth", val, 100);
            return val;
        }
        set 
        {
            SetGV("UTAddOn_save_maxHealth", value);
        }
    }

    // Max heal armor
    // --------------
    int maxHealArmor
    {
        get const 
        {
            int val;
            GetGV("UTAddOn_save_maxArmor", val, 5);
            return val;
        }
        set 
        {
            SetGV("UTAddOn_save_maxArmor", value);
        }
    }

    // Difficulty
    // ----------
    int difficulty
    {
        get const 
        {
            int val;
            GetGV("UTAddOn_save_difficulty", val, UTA_DIFF_NORMAL);
            return val;
        }
        set 
        {
            SetGV("UTAddOn_save_difficulty", value);
        }
    }

    // FirstBlood
    // ----------
    int firstBloodSpilled
    {
        get const 
        {
            int val;
            GetGV("UTAddOn_save_firstBlood", val, 0);
            return val;
        }
        set 
        {
            SetGV("UTAddOn_save_firstBlood", value);
        }
    }

    // SessionID
    // ---------
    int sessionID
    {
        get const 
        {
            int val;
            GetGV("UTAddOn_save_sessionID", val, 0);
            return val;
        }
        set 
        {
            SetGV("UTAddOn_save_sessionID", value);
        }
    }

    // KillingSpree Counter
    // --------------------
    int killSpreeCounter
    {
        get const 
        {
            int val;
            GetGV("UTAddOn_save_killSpreeCounter", val, 0);
            return val;
        }
        set 
        {
            SetGV("UTAddOn_save_killSpreeCounter", value);
        }
    }

    // KillingSpree Remainder
    // ----------------------
    int killSpreeRemainder
    {
        get const 
        {
            int val;
            GetGV("UTAddOn_save_killSpreeRemainder", val, 0);
            return val;
        }
        set 
        {
            SetGV("UTAddOn_save_killSpreeRemainder", value);
        }
    }

    // KillingSpree nKillsBegin
    // ------------------------
    int killSpreeKillsBegin
    {
        get const 
        {
            int val;
            GetGV("UTAddOn_save_killSpreeKillsBegin", val, 0);
            return val;
        }
        set 
        {
            SetGV("UTAddOn_save_killSpreeKillsBegin", value);
        }
    }
    
    

    // KillStreakCounter Array
    // -----------------------
    array<int> killStreakCounter
    {
        get const
        {
            array<int> ksArray(9, 0);
            for(int i = 0; i < 9; i++)
            {
                int val;
                GetGV("UTAddOn_save_killcount_" + (i + 1), val, 0);
                ksArray[i] = val;
            }
            return ksArray;
        }
        set
        {
            for(int i = 0; i < 9; i++)
            {
                SetGV("UTAddOn_save_killcount_" + (i + 1), value[i]);
            }
        }
    }
    

    // ===============================================


    void InitSaveData()
    {
        int value;

        // Load all game variables and create those which don't already exist
        // -> The property getters (using GetGV) will already account for this, so we just abuse this
        value = maxHealHealth;
        value = maxHealArmor;
        value = difficulty;
        value = firstBloodSpilled;
        value = sessionID;
        value = killSpreeCounter;
        value = killSpreeRemainder;
        value = killSpreeKillsBegin;
        array<int> ksArrayInit = killStreakCounter;
        g_killCounter = ksArrayInit[0];

        Warn("Loaded/Added GameVariables:");
        Warn("- sessionID: " + sessionID);
        Warn("- maxHealth: " + maxHealHealth);
        Warn("- maxArmor: " + maxHealArmor);
        Warn("- difficulty: " + difficulty);
        Warn("- firstBlood: " + firstBloodSpilled);
        Warn("- SpreeKillsBegin: " + killSpreeKillsBegin);
        Warn("- SpreeCount: " + killSpreeCounter);
        Warn("- SpreeRemainder: " + killSpreeRemainder);

        kStr strDbgSpree = "- KillStreaks: ";
        for(int i = 0; i < 9; i++)
        {
            strDbgSpree += "" + ksArrayInit[i];
            if(i < 8)
            {
                strDbgSpree += ", ";
            }
        }
        Warn(strDbgSpree);

    }


    void LoadIntArray(array<int> &inout intAr, kStr strVarPrefix, kStr strDebugInfo)
    {
        int nElem = intAr.length();
        kStr strDebug = strDebugInfo + ": ";
        for(int i = 0; i < nElem; i++)
        {
            kStr varName = strVarPrefix + (i + 1);
            GameVariables.GetInt(varName, intAr[i], 0); // Default value must(!) be hard-coded to some value (so zero)
            strDebug += "" + intAr[i];
            if(i + 1 < nElem)
            {
                strDebug += ", ";
            }
        }
        Warn(strDebug);
    }
    void SaveIntArray(array<int> &inout intAr, kStr strVarPrefix, kStr strDebugInfo)
    {
        int nElem = intAr.length();
        kStr strDebug = strDebugInfo + ": ";
        for(int i = 0; i < nElem; i++)
        {
            kStr varName = strVarPrefix + (i + 1);
            GameVariables.SetValue(varName, "" + intAr[i]);
            strDebug += "" + intAr[i];
            if(i + 1 < nElem)
            {
                strDebug += ", ";
            }
        }
        Warn(strDebug);
    }

}
UT_SaveData g_saveData; // Global variable for save data storage


// This function must be called to correctly change the UTAddOn difficulty
void ChangeDifficultySetting(int difficulty)
{
    switch(difficulty)
    {
        case UTA_DIFF_OFF:
            g_gamePlaySettings.enableDamage = false;
            g_gamePlaySettings.enableHealing = false;
            g_gamePlaySettings.enableArmorHealing = false;
            g_gamePlaySettings.enableAchievements = false;
            g_gamePlaySettings.maxHealRate = 1.0f;  // value doesn't matter
            g_gamePlaySettings.streakDuration = 2.5f;
            g_gamePlaySettings.enableDamageInterrupt = true;    // has no effect
            break;
        case UTA_DIFF_BABY:
            g_gamePlaySettings.enableDamage = false;
            g_gamePlaySettings.enableHealing = true;
            g_gamePlaySettings.enableArmorHealing = true;
            g_gamePlaySettings.enableAchievements = true;
            g_gamePlaySettings.maxHealRate = 0.0f;  // unlimited
            g_gamePlaySettings.streakDuration = 3.5f;
            g_gamePlaySettings.enableDamageInterrupt = true;
            break;
        case UTA_DIFF_EASY:
            g_gamePlaySettings.enableDamage = true;
            g_gamePlaySettings.enableHealing = true;
            g_gamePlaySettings.enableArmorHealing = true;
            g_gamePlaySettings.enableAchievements = true;
            g_gamePlaySettings.maxHealRate = 1.0f / 10.0f;
            g_gamePlaySettings.streakDuration = 3.0f;
            g_gamePlaySettings.enableDamageInterrupt = true;
            break;
        case UTA_DIFF_NORMAL:
            g_gamePlaySettings.enableDamage = true;
            g_gamePlaySettings.enableHealing = true;
            g_gamePlaySettings.enableArmorHealing = false;
            g_gamePlaySettings.enableAchievements = true;
            g_gamePlaySettings.maxHealRate = 1.0f / 5.0f;
            g_gamePlaySettings.streakDuration = 2.5f;
            g_gamePlaySettings.enableDamageInterrupt = true;
            break;
        case UTA_DIFF_HARD:
            g_gamePlaySettings.enableDamage = true;
            g_gamePlaySettings.enableHealing = true;
            g_gamePlaySettings.enableArmorHealing = false;
            g_gamePlaySettings.enableAchievements = true;
            g_gamePlaySettings.maxHealRate = 1.0f / 3.0f;
            g_gamePlaySettings.streakDuration = 2.5f;
            g_gamePlaySettings.enableDamageInterrupt = false;
            break;
        case UTA_DIFF_HARDCORE:
            g_gamePlaySettings.enableDamage = true;
            g_gamePlaySettings.enableHealing = false;
            g_gamePlaySettings.enableArmorHealing = false;
            g_gamePlaySettings.enableAchievements = false;
            g_gamePlaySettings.maxHealRate = 10.0f; // off
            g_gamePlaySettings.streakDuration = 2.5f;
            g_gamePlaySettings.enableDamageInterrupt = false;
            break;
        default:
            Warn("Unexpected difficulty: " + difficulty);
    }

    g_saveData.difficulty = difficulty;
}

}   // NS: UTAddOn
