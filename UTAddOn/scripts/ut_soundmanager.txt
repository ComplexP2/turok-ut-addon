class UT_SoundData

}

class UT_SoundDict  // frak kDict
{
    kStr strCOMP = "Sound::SoundDict";

    array<int> keys;
    array<UT_SoundData> values;

    UT_SoundDict()
    {
        Clear();
    }

    void Clear()
    {
        keys.resize(0);
        values.resize(0);
    }

    int Length()
    {
        return keys.length();
    }

    void Add(int key, UT_SoundData value)
    {
        int idxElem = keys.find(key);
        if(idxElem < 0)
        {
            keys.insertLast(key);
            values.insertLast(value);
        }
        else
        {
            Warn(strCOMP, "Key '" + key + "' already exists!");
        }
    }

    UT_SoundData Get(int key)
    {
        int idxElem = keys.find(key);
        if(idxElem >= 0)
        {
            return values[idxElem];
        }
        else
        {
            Warn(strCOMP, "Key '" + key + "' not found!");
            return UT_SoundData();
        }
    }

}

    void RegisterSound(int soundId, int soundCat, kStr filename, float duration_s)
    {
        UT_SoundData data;
        data.soundId = soundId;
        data.soundCategory = soundCat;
        data.filename = filename;
        data.duration_ticks = int(duration_s * 60.0f);

        dict_snd.Add(soundId, data);
    }

    void Tick(bool isOnOwnageStreak)
    {
        // Update timers
        if(timer_killStreak > 0)
        {
            timer_killStreak--;
        }
        if(timer_ownageStreak > 0)
        {
            timer_ownageStreak--;
        }
        if(timer_event > 0)
        {
            timer_event--;
        }
        if(timer_special > 0)
        {
            timer_special--;
        }
        if(timer_killSpree > 0)
        {
            timer_killSpree--;
        }

        // Play sounds stored in queues
        if(timer_ownageStreak == 0 && timer_killStreak == 0)    // > 0 means a sound is playing
        {
            // Play only if killstreak sound has finished

            if(queue_Event.Size() > 0)
            {
                // Play event sounds of queue

                int soundId = -1;
                queue_Event.Pop(soundId);

                UT_SoundData data = dict_snd.Get(soundId);
                if(data.soundId < 0)
                {
                    Warn(strCOMP, "Unexpected: Key not found: " + soundId);
                    return;
                }

                PlaySound_Event(data.filename, data.duration_ticks);
            }
            else if(timer_event == 0)
            {

                // Play OwnageStreak sounds only if event sounds have finished and
                // if no killing streak sound is currently playing
                // (and wait for other killspree sounds as well)

                if(queue_OwnageStreak.Size() > 0 && timer_killSpree == 0 && 
                   timer_ownageStreak == 0)
                {
                    int soundId;
                    if(queue_OwnageStreak.Pop(soundId))
                    {
                        UT_SoundData data = dict_snd.Get(soundId);
                        if(data.soundId < 0)
                        {
                            Warn(strCOMP, "Unexpected: Key not found: " + soundId);
                            return;
                        }
                        Log("PlaySound: " + data.filename);
                        PlaySound_OwnageStreak(data.filename, data.duration_ticks);
                    }
                }


                // Start playing a KillSpree sound only, if the ownage streak queue
                // is empty and no ownage streak sound is currently playing and player
                // is not on an ownage streak right now
                // -> prevents KillSpree sounds playing prior to OwnageStreak sounds
                // (and wait for other killspree sounds as well)

                if(!isOnOwnageStreak &&
                   queue_KillSpree.Size() > 0 && timer_killSpree == 0 && 
                   queue_OwnageStreak.Size() == 0 && timer_ownageStreak == 0)
                {
                    // Play killspree sounds of queue

                    int soundId;
                    if(queue_KillSpree.Pop(soundId))
                    {
                        UT_SoundData data = dict_snd.Get(soundId);
                        if(data.soundId < 0)
                        {
                            Warn(strCOMP, "Unexpected: Key not found: " + soundId);
                            return;
                        }
                        Log("PlaySound: " + data.filename);
                        PlaySound_KillSpree(data.filename, data.duration_ticks);
                    }
                }

            }
        }
    }

    void Request_PlaySound(int soundId)
    {
        UT_SoundData data = dict_snd.Get(soundId);
        if(data.soundId < 0)
        {
            Warn(strCOMP, "Unexpected: Key not found: " + soundId);
            return;
        }

        if(timer_special > 0 && data.soundCategory != UT_SND_CAT_ABOVE_SPECIAL)
        {
            Log(strCOMP, "Sound with id " + soundId + " supressed by special sound");
            return;
        }

        switch(data.soundCategory)
        {
            case UT_SND_CAT_SPECIAL:
                // Play sound immediately
                PlaySound(data.filename);
                // Set special timer
                timer_special = data.duration_ticks;
                break;

            case UT_SND_CAT_KILLSTREAK:
                // Play sound immediately and set timer
                // -> But only if no special sound is currently running
                PlaySound_KillStreak(data.filename, data.duration_ticks);
                break;

            case UT_SND_CAT_KILLSPREE:
                // Put sound into queue for later playback
                queue_KillSpree.Push(soundId, true);
                break;

            case UT_SND_CAT_OWNAGESTREAK:
                queue_OwnageStreak.Push(soundId, true);
                break;

            case UT_SND_CAT_EVENT:
                // Put sound into queue for later playback
                queue_Event.Push(soundId);
                break;

            case UT_SND_CAT_PLAYER:
            case UT_SND_CAT_ABOVE_SPECIAL:
                // Play sound immediately
                PlaySound(data.filename);
                break;

            default:
                Warn(strCOMP, "Unknown sound category: " + data.soundCategory);
        }

    }

    void ReduceSpreeQueueToHighest() {
        
        array<int> data = queue_KillSpree.GetArray();
        int nData = data.length();

        if(nData == 0) 
        {
            return;
        }
        
        // Find max value
        int maxValue = -1;
        for(int i = 0; i < nData; i++)
        {
            if(data[i] > maxValue) {
                maxValue = data[i];
            }
        }

        if(maxValue < 0)
        {
            return;
        }

        // Reduce queue to one maxValue entry
        array<int> newData;
        newData.insertAt(0, maxValue);

        queue_KillSpree.SetArray(newData);
    }

    {

    }

        Game.PlaySound(filename);

    }

    void PlaySound_KillSpree(kStr filename, int nTicks)
    {
        Game.PlaySound(filename);
        timer_killSpree = nTicks;
    }

    void PlaySound_OwnageStreak(kStr filename, int nTicks)
    {
        Game.PlaySound(filename);
        timer_ownageStreak = nTicks;
    }

    void PlaySound_Event(kStr filename, int nTicks)
    {
        Game.PlaySound(filename);
        timer_event = nTicks;
    }
