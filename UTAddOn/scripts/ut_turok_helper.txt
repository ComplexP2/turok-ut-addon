bool IsInGame(int levelId)
{
	return (levelId != UTA_LVL_ID_CREDITS && levelId != UTA_LVL_ID_NONE);
}

bool IsBossFight(int levelId)
{
	return (levelId == UTA_LVL_ID_LEVEL_3_BOSS || levelId == UTA_LVL_ID_LEVEL_5_BOSS ||
	        levelId == UTA_LVL_ID_LEVEL_8_BOSS_1 || levelId == UTA_LVL_ID_LEVEL_8_BOSS_2);
}

bool IsInSpiritWorld(int levelId)
{
	return (levelId <= UTA_LVL_ID_SPIRIT_LEVEL_1 && levelId >= UTA_LVL_ID_SPIRIT_LEVEL_8);
}

bool IsTraining(int levelId)
{
	return (levelId == UTA_LVL_ID_LEVEL_TRAINING);
}

bool IsCredits(int levelId)
{
	return (levelId == UTA_LVL_ID_CREDITS);
}


// ==========
// GetLevelId
// ==========
// Convert the map id to the corresponding level number.
// Unknown map ids will get an offset of 100 to prevent conflicts.
// Although all paths return a value, the compiler complains if the final return is not set. What the frak ...
int GetLevelId()
{
    int mapId = Game.GetCurrentMapID();

    switch(mapId)
    {
		// Not an active game (menu, sequences)
		case 1:		// Escape from Campaigner base
        case 42:	// Menu
		case 43:
		case 44:
		case 45:
		case 46:
            return UTA_LVL_ID_NONE;

		// Credits
		case 2:
			return UTA_LVL_ID_CREDITS;

		// Training
		case 47:
			return UTA_LVL_ID_LEVEL_TRAINING;

		// Spirit stuff
		case 26:
		case 27:
			return UTA_LVL_ID_SPIRIT_LEVEL_1;
		case 28:
		case 29:
			return UTA_LVL_ID_SPIRIT_LEVEL_2;
		case 30:
		case 31:
			return UTA_LVL_ID_SPIRIT_LEVEL_3;
		case 32:
		case 33:
			return UTA_LVL_ID_SPIRIT_LEVEL_4;
		case 34:
		case 35:
			return UTA_LVL_ID_SPIRIT_LEVEL_5;
		case 36:
		case 37:
			return UTA_LVL_ID_SPIRIT_LEVEL_6;
		case 38:
		case 39:
			return UTA_LVL_ID_SPIRIT_LEVEL_7;
		case 40:
		case 41:
			return UTA_LVL_ID_SPIRIT_LEVEL_8;

        // Level 1
		case 4:
		case 5:     // Includes hub
			return UTA_LVL_ID_LEVEL_1;

        // Level 2
		case 6:
		case 7:
			return UTA_LVL_ID_LEVEL_2;

        // Level 3
		case 8:
		case 9:
			return UTA_LVL_ID_LEVEL_3;
		case 48:    // Boss: 2 Hummer + Longhunter
			return UTA_LVL_ID_LEVEL_3_BOSS;

        // Level 4
		case 10:
		case 11:
			return UTA_LVL_ID_LEVEL_4;

        // Level 5
		case 12:
		case 13:
			return UTA_LVL_ID_LEVEL_5;
		case 49:    // Boss: Mantis
			return UTA_LVL_ID_LEVEL_5_BOSS;

        // Level 6
		case 14:
		case 15:
		case 16:
			return UTA_LVL_ID_LEVEL_6;

        // Level 7
		case 17:
		case 18:
		case 19:
		case 20:
			return UTA_LVL_ID_LEVEL_7;

        // Level 8
		case 21:
		case 22:
		case 23:
		case 24:
		case 25:	// Pre-Campaigner hallway
			return UTA_LVL_ID_LEVEL_8;
		case 3:     // Boss: T-Rex
			return UTA_LVL_ID_LEVEL_8_BOSS_1;
		case 0:     // Boss: Campaigner
			return UTA_LVL_ID_LEVEL_8_BOSS_2;
				

        // Custom/Unknown
        default:
            return UTA_LVL_ID_OFFSET_CUSTOM + mapId;
    }

    return UTA_LVL_ID_NONE;
}


// ===============
// DetectBossFight
// ===============
// Detect if a boss currently exists and return the boss actor handle
bool DetectBossFight(kActor@ &out pBoss)
{

	@pBoss = World.GetActorByTID(441, AT_AIBOSS_MANTIS);
	if(@pBoss is null)
	{
		@pBoss = World.GetActorByTID(400, AT_AIBOSS_HUMMER);
	}
	if(@pBoss is null)
	{
		@pBoss = World.GetActorByTID(400, AT_AIBOSS_HUNTER);
		if(pBoss !is null)
		{
			kActor @hummer1 = World.GetActorByTID(1140);
			kActor @hummer2 = World.GetActorByTID(1141);
			if(hummer1.Health() > 0)
			{
				@pBoss = hummer1;
			}
			else if(hummer2.Health() > 0)
			{
				@pBoss = hummer2;
			}
		}
	}
	if(@pBoss is null)
	{
		@pBoss = World.GetActorByTID(400, AT_AIBOSS_CAMPAINGER);
	}
	if(@pBoss is null)
	{
		@pBoss = World.GetActorByTID(400, AT_AIBOSS_TREX);
	}
	
	if(@pBoss !is null)
	{
		return true;
	}
	else 
	{
		return false;
	}

	// Don't return something here, because all paths already return a value and for some reason this would kill the compiler lol ...
}

// ================
// SpawnItemOnActor
// ================
// Spawn an item originating from the given actor
UT_TemporaryActor SpawnItemOnActor(kActor @pActor, kStr strItem, float lifeTime_s)
{
	kVec3 pos = pActor.Origin();
	int secIdx = pActor.GetSectorIndexAtLocation(pos, 0);

	return SpawnItemAtPosition(pos, strItem, lifeTime_s, secIdx);
}

// ===================
// SpawnItemAtPosition
// ===================
// Spawn an item originating from the given position
UT_TemporaryActor SpawnItemAtPosition(kVec3 pos, kStr strItem, float lifeTime_s, int secIdx)
{

	kActor @item = ActorFactory.Spawn(strItem,
										pos.x,
										pos.y,
										pos.z + 32.0f,
										0, secIdx);
										
	item.Gravity() = 0.5f;
	item.BounceDamp() = 0.5f;
	item.ClipFlags() = (CF_DROPOFF|CF_CLIPEDGES|CF_NOCLIPACTORS|CF_COLLIDEFLOORS|CF_COLLIDEHEIGHT);
	item.Velocity().x = Math::RandCFloat() * 4.096f;
	item.Velocity().y = Math::RandCFloat() * 4.096f;
	item.Velocity().z = 4.096f + (Math::RandFloat() * 4.096f);
	
	item.RunFxEvent("Item_Spawn");

	// Actor with limited lifetime
	UT_TemporaryActor tempActor;
	@tempActor.actorRef = item;
	tempActor.lifeTime_s = lifeTime_s;
	tempActor.despawnFxId = 1;

	return tempActor;
}


// ==============
// SpawnLifeForce
// ==============
// Spawns a lifeforce object into Turok (so it's collected immediately)
UT_TemporaryActor SpawnLifeForce()
{
	Log("Spawning: UTA_LifeForce_2_Silent");
	
	kVec3 pos = Player.Actor().Origin();
	int secIdx = Player.Actor().SectorIndex();
	kActor @item = ActorFactory.Spawn("UTA_LifeForce_2_Silent",
										pos.x,
										pos.y,
										pos.z + Player.Actor().ViewHeight()/2.0f,
										0, secIdx);

	item.Scale() = kVec3(0.01f, 0.01f, 0.01f);

	// Actor with limited lifetime
	UT_TemporaryActor tempActor;
	@tempActor.actorRef = item;
	tempActor.lifeTime_s = 4.0f / 60.0f;	// 4 Ticks
	tempActor.despawnFxId = 0;

	return tempActor;
}


class WeaponInfo
{
	int weaponIndex = -1;
	bool ownsWeapon = false;
	bool ownsAmmo = false;
	bool ownsAltAmmo = false;
	bool hasAltAmmo = false;	// indicates if the weapon has an alt ammo slot
	int ammo = 0;
	int altAmmo = 0;
	int maxAmmo = 0;
	int maxAltAmmo = 0;
	float pcAmmo = 0.0f;		// ratio of current ammo to max ammo
	float pcAltAmmo = 0.0f;		// ratio of current alt ammo to max alt ammo
}


WeaponInfo GetWeaponInfo(int idxWp)
{
	WeaponInfo wpInfo;

	wpInfo.weaponIndex = idxWp;

	wpInfo.ownsWeapon = Player.HasWeapon(idxWp);
	wpInfo.ammo = Player.GetAmmo(idxWp);
	wpInfo.altAmmo = Player.GetAltAmmo(idxWp);

	wpInfo.ownsAmmo = (wpInfo.ammo > 0);
	wpInfo.ownsAltAmmo = (wpInfo.altAmmo > 0);

	wpInfo.maxAmmo = g_maxAmmo[idxWp];
	wpInfo.maxAltAmmo = g_maxAltAmmo[idxWp];

	if(wpInfo.ammo > 0)
	{
		if(wpInfo.ammo > g_maxAmmo[idxWp])
		{
			// Unexpected ammo amount (probably because of another mod) 
			// -> increase max value automatically
			g_maxAmmo[idxWp] = wpInfo.ammo;
			wpInfo.maxAmmo = wpInfo.ammo;
			Log("Max ammo for weapon " + idxWp + " has been increased to " + wpInfo.ammo);
		}
		wpInfo.pcAmmo = float(Clip(wpInfo.ammo, 0, wpInfo.maxAmmo)) / float(wpInfo.maxAmmo);
	}
	
	if(wpInfo.altAmmo > 0)
	{
		if(wpInfo.altAmmo > g_maxAltAmmo[idxWp])
		{
			// Unexpected ammo amount (probably because of another mod) 
			// -> increase max value automatically
			g_maxAltAmmo[idxWp] = wpInfo.altAmmo;
			wpInfo.maxAltAmmo = wpInfo.altAmmo;
			Log("Max alt ammo for weapon " + idxWp + " has been increased to " + wpInfo.altAmmo);
		}
		wpInfo.pcAltAmmo = float(Clip(wpInfo.altAmmo, 0, wpInfo.maxAltAmmo)) / float(wpInfo.maxAltAmmo);
	}

	wpInfo.hasAltAmmo = (wpInfo.maxAltAmmo > 0);

	return wpInfo;
}
