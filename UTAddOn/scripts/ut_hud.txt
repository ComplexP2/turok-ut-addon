// ==========
// HUD Mapper
// ==========
// Computes world coordinates and orientations for plane objects
// which are specified in HUD coordinates

const float facHudCoord = 1000.0f;  // Number in "hud coordinates" which corresponds to 50% "screen size"

// HUD Coordinate system:
//
// --------------------------------------------------------
// |                     facHudCoord (y+)                 |
// |                           |                          |
// |                           |                          |
// |                           |                          |
// | -facHudCoord (x-) ------- 0 ------- facHudCoord (x+) |
// |                           |                          |
// |                           |                          |
// |                           |                          |
// |                    -facHudCoord (y-)                 |
// --------------------------------------------------------
//
// Note: width/height ratio is applied in helper functions

class UT_Hud_Mapper
{
    kStr strCOMP = "Hud::Mapper";

    kVec3 basePos_;
    kQuat baseRot_;
    kQuat climbRot_ = kQuat(0, ex);
    UT_Hud_VideoSettings videoSettings_;

    UT_Hud_Mapper()
    {
    }

    void Init()
    {
        GenerateVideoSettings();
    }

    void GenerateVideoSettings()
    {
        // Get FOV
        kStr strFOV;
        if(Sys.GetCvarValue("r_fov", strFOV))
        {
            videoSettings_.fov = Math::Deg2Rad(strFOV.Atof());
        }
        else
        {
            videoSettings_.fov = Math::Deg2Rad(74.0f);
        }

        // Get screen resolution setting
        videoSettings_.screenW = Sys.VideoWidth();
        videoSettings_.screenH = Sys.VideoHeight();

        // Compute helper values
        videoSettings_.ratio_W_H = float(videoSettings_.screenW) / float(videoSettings_.screenH);
        videoSettings_.ratio_H_z = 2 * Math::Tan(videoSettings_.fov / 2.0f);
        // Note: z-value * ratio_H_z gives the vertical screen size in "game units" (z-value is in "game units")
    }

    UT_Hud_VideoSettings GetVideoSettings()
    {
        return videoSettings_;
    }


    // Compute the camera origin position and orientation
    void PreComputeHudPosParams()
    {
        // Compute player actor dependent hud origin position and orientation

        kPuppet @pPuppet = Player.Actor();

        // Orientation in "look forward" direction
        baseRot_ = pPuppet.Rotation() * kQuat(pPuppet.RecoilPitch(), ex);

        // Camera origin position in world coords
        float zOffset = pPuppet.ViewHeight() + pPuppet.LandingViewOffset(); // Originate from eye position
        basePos_ = pPuppet.Origin() + ez * zOffset;
    }

    // Compute the object scale in world coordinates based on the given 
    // HUD coordinates and z-distance
    kVec3 ScreenScaleToWorldScale(float z, float hudW, float hudH)
    {
        // hudW/H = 2*facHudCoord * percentage of screen size
        // z is already given in world units and is interpreted as z-position in cam coordinates

        kVec3 scale;

        float zOff = z + hud_zOff;
       
        float sf = zOff * videoSettings_.ratio_H_z / facHudCoord / 2.0f;

        scale.x = sf * hudW * videoSettings_.ratio_W_H;
        scale.y = sf * hudH;
        scale.z = 1.0f;

        return scale;
    }

    // Compute the position in camera coordinates of an object 
    // based on it's HUD coordinates and z-distance
    kVec3 ScreenPosToCamPos(float z, float hudX, float hudY)
    {
        // hudX/Y = hud coordinates = 2*facHudCoord * percentage of screen size
        // z is interpreted as z-position in cam coordinates in world units

        kVec3 camPos;

        float zOff = z + hud_zOff;

        float sf = zOff * videoSettings_.ratio_H_z / facHudCoord / 2.0f;

        camPos.x = sf * hudX * videoSettings_.ratio_W_H;
        camPos.z = sf * hudY;
        camPos.y = zOff;

        return camPos;
    }


    // Position the given actor in the world and compute it's orientation,
    // based on the given HUD coordinates, scales and z-distance
    void PositionHudElement(kActor @elem, float z, float hudX, float hudY, float hudW, float hudH, kQuat elemRot)
    {
        kVec3 cpos = ScreenPosToCamPos(z, hudX, hudY);
        kVec3 scale = ScreenScaleToWorldScale(z, hudW, hudH);

        PositionHUDActor(elem, cpos, elemRot);
        elem.Scale().Set(scale.x*fac,scale.y*fac, scale.z);
    }


    // OLD positioning function!
    // TODO: Refactor all references and then move the code of this function
    //       to PositionHudElement
    void PositionHUDActor(kActor @actor, kVec3 camPos, kQuat rot)
    {
        kVec3 vDir = camPos * climbRot_ * baseRot_;
        actor.SetPosition(basePos_ + vDir);
        actor.Rotation() = Player.Actor().Rotation() * climbRot_ * rot;
    }
    // TODO: Refactor references to this function and then delete it!
    kVec3 HudPositionToWorldPosition(kVec3 hudPosition)
    {
        kVec3 vDir = hudPosition * baseRot_;
        return basePos_ + vDir;
    }

}

// Generic class to display an N-digit number
class UT_Hud_Numberplate
{
    kStr strCOMP = "Hud::Numberplate";

    int nDigits_ = 0;
    array<kActor@> actors_;

    UT_Hud_Numberplate(int digits)
    {
        nDigits_ = digits;
    }

    void Init()
    {
        if(nDigits_ < 1)
        {
            Warn(strCOMP, "Numberplate with zero digits is ignored");
            return;
        }

        actors_.resize(0);
        for(int i = 0; i < nDigits_; i++)
        {
            kActor @temp = ActorFactory.Spawn("UTA_Generic_Quad_Plane_GFX", 0,0,0, 0, Player.Actor().SectorIndex());
            temp.Scale().Set(fac, fac, 1.0f);
            actors_.insertLast(temp);
        }
    }

    void Invalidate()
    {
        actors_.resize(0);
    }

    void Update(int number, float gap, float z, float hudX, float hudY, float hudW, float hudH, UT_Hud_Mapper &HMap)
    {
        // Hard-coded: align left and center-y
        float DW = hudW / nDigits_;

        int num = Max(0, number);
        for(int i = 0; i < nDigits_; i++)
        {
            kActor @temp = actors_[i];
            // hudX + DW/2 | + DW + gap | + DW + gap | ...
            // -> reversed order: (nDigits_ - 1 - i)
            float posDX = hudX + DW / 2.0f + (nDigits_ - 1 - i) * (DW + gap);

            int digit = (num % 10);
            num = (num - digit) / 10;
            if(digit == 0 && num == 0 && i > 0)
            {
                // Don't show leading zeros (except one)
                temp.RenderModel().SetVisibility(0, false);
            }
            else
            {
                // Show digit
                HMap.PositionHudElement(temp, z, posDX, hudY, DW, hudH, generic_plane_rot);
                temp.RenderModel().SetTexture(0, tex_offset_gfx_Numbers + digit);
                temp.RenderModel().SetVisibility(0, true);
            }
        }
    }

}

        float posXBg = posX0 + bgElemW / 2.0f;

        float maxHealth = float(hud_data_.health_max);
        float maxArmor = float(hud_data_.armor_max);
        float barMaxW = NumW;
        float barW = barMaxW;   // to be overridden
        float barH = NumH / 4.0f;
        float barPosX0 = posXNm;
        float barPosY0 = posY0 - bgElemH / 2.0f + barH * 2.0f;
        float barPosX = barPosX0;   // to be overridden

        // ==============

        // [0] Background image

                                                posXNm, posY0,

                                barPosX, barPosY0, 
                                barW, barH, generic_plane_rot);
        temp.RenderModel().SetTexture(0, tex_offset_gfx_HudBars);

        barPosX = barPosX0 + barW / 2.0f;
        @temp = elems_healthDisp_.actors[3];
        HMap_.PositionHudElement(temp, pz_nm, 
                                barPosX, barPosY0 - barH, 
                                barW, barH, generic_plane_rot);
        temp.RenderModel().SetTexture(0, tex_offset_gfx_HudBars + 1);

        temp.RenderModel().SetTexture(0, tex_offset_gfx_HudBars + 2);

        barPosX += barW / 2.0f;
        barW = barMaxW * (float(hud_data_.health_max) - float(hud_data_.health_heal)) / maxHealth;
        barPosX += barW / 2.0f;

                                barW, barH, generic_plane_rot);
        temp.RenderModel().SetTexture(0, tex_offset_gfx_HudBars + 3);



        // Armor display
        // =============

        // Shift stuff upwards
        posY0 += bgElemH * 1.1f;
        barPosY0 = posY0 - bgElemH / 2.0f + barH * 2.0f;

        // [0] Background image
        @temp = elems_armorDisp_.actors[0];
        HMap_.PositionHudElement(temp, pz_bg, posXBg, posY0, bgElemW, bgElemH, generic_plane_rot);

        // [1] Armorshield image
        @temp = elems_armorDisp_.actors[1];

        // [-] Armor number display
        elem_num_armor_.Update(hud_data_.armor, -10.0f, pz_nm,

                                                NumW, NumH, HMap_);

        // [2] Yellow armor bar (current armor)
        barW = barMaxW * float(Clip(hud_data_.armor, 0, hud_data_.armor_max)) / maxArmor;
        barPosX = barPosX0 + barW / 2.0f;
        @temp = elems_armorDisp_.actors[2];

    void Update_UI_WeaponGFX()
    {
        if(elems_weaponGFX_.actors.length() != nElems_weaponGFX)
        {
            Warn(strCOMP, "Unexpected elems_weaponGFX_ array size");
            return;
        }
        
        if(elems_weaponBgGFX_.actors.length() != nElems_weaponGFX)
        {
            Warn(strCOMP, "Unexpected elems_weaponBgGFX_ array size");
            return;
        }

        if(elems_weaponBgStackGFX_.actors.length() != nElems_weaponsStacked)
        {
            Warn(strCOMP, "Unexpected elems_weaponBgStackGFX_ array size");
            return;
        }

        if(elems_weaponAmmoBar_.actors.length() != nElems_weaponAmmo)
        {
            Warn(strCOMP, "Unexpected elems_weaponAmmoBar_ array size");
            return;
        }

        if(elems_weaponAltAmmoBar_.actors.length() != nElems_weaponAltAmmo)
        {
            Warn(strCOMP, "Unexpected elems_weaponAltAmmoBar_ array size");
            return;
        }

        kActor @temp = null;
        kActor @tempBg = null;
        kActor @tempSt = null;
        kActor @tempBar = null;
        kActor @tempAltBar = null;

        float pz_wp = 1.125f;
        float pz_am = 1.0f;
        float pz_st = 1.25f;
        float pz_bg = 1.5f;

        float wpHudElemWidth = 150.0f;
        float wpHudElemHeight = wpHudElemWidth * 0.9f;
        float wpHudBgToWpScale = 0.75f;
        float wpHudStackYOff = wpHudElemHeight * 0.98f;
        float yOffDoubleSlot = 4.0f * (wpHudElemHeight - wpHudStackYOff);

        float posX0 = -900.0f;
        float posY0 = -750.0f;        

        // Iterate over all slots
        int curWeaponIdx = Player.CurrentWeapon();
        int curSlotIdx = weaponSlotMapping[curWeaponIdx];
        for(int iSlot = 0; iSlot < nElems_weaponGFX; iSlot++)
        {
            float pz_off = 10.0f;
            if(iSlot > 5)
            {
                pz_off = 0.0f;
            }

            // Weapon ownage and slot type
            bool ownsSlotWeapon = Player.HasWeapon(slotWeaponMapping1[iSlot]);
            bool ownsAltSlotWeapon = Player.HasWeapon(slotWeaponMapping2[iSlot]);
            bool isDoubleSlot = (slotStackID[iSlot] > 0);

            // Detect selected slot and corresponding weapon (texture)
            // ---------------------------------------------
            int idxWpTex = 0;   // Weapon texture index to display
            int idxWp = 0;      // Index of weapon that is displayed
            bool isCurSlotSelected = false;
            bool isAltSlotSelected = false;

            if(iSlot == curSlotIdx)
            {
                // Current slot has the current weapon -> is selected
                isCurSlotSelected = true;
                idxWp = curWeaponIdx;
                idxWpTex = curWeaponIdx;
                isAltSlotSelected = (slotWeaponMapping1[iSlot] != curWeaponIdx);
            }
            else
            {
                // Current slot is not selected
                if(ownsSlotWeapon)
                {
                    idxWp = slotWeaponMapping1[iSlot];
                    idxWpTex = idxWp;
                }
                else if(ownsAltSlotWeapon)
                {
                    idxWp = slotWeaponMapping2[iSlot];
                    idxWpTex = idxWp;
                }
                else
                {
                    idxWp = slotWeaponMapping1[iSlot];
                    idxWpTex = idxWp + tex_offset_gfx_WeaponBlack;
                }
            }

            /*if(idxWp == 0)
            {
                idxWp = 1;
            }*/
            

            // Position weapon background element
            // ----------------------------------
            float posBgX = posX0 + wpHudElemWidth * (float(iSlot) + 0.5f);
            float posBgY = posY0;

            int tex_idx = tex_offset_gfx_HudElements;   // Default background
            if(isCurSlotSelected)
            {
                tex_idx = tex_offset_gfx_HudElements + 3;   // Selected background
            }
            if(isAltSlotSelected)
            {
                posBgY += yOffDoubleSlot;   // Shift upwards
            }
            @tempBg = elems_weaponBgGFX_.actors[iSlot];
            HMap_.PositionHudElement(tempBg, pz_bg + pz_off, posBgX, posBgY, wpHudElemWidth, wpHudElemHeight, generic_plane_rot);
            tempBg.RenderModel().SetTexture(0, tex_idx);
            

            // Position weapon background topper/bottomer element
            // --------------------------------------------------
            if(isDoubleSlot)
            {
                bool showTopper = false;
                bool showBottomer = false;

                if(isAltSlotSelected)
                {
                    if(ownsSlotWeapon)
                    {
                        showBottomer = true;
                    }
                }
                else if(ownsSlotWeapon && ownsAltSlotWeapon)
                {
                    showTopper = true;
                }

                int idx = slotStackID[iSlot] - 1;
                @tempSt = elems_weaponBgStackGFX_.actors[idx];
                if(showBottomer || showTopper)
                {
                    float posYSt = posBgY;
                    if(showTopper)
                    {
                        posYSt = posBgY + wpHudStackYOff;
                        tempSt.RenderModel().SetTexture(0, tex_offset_gfx_HudElements + 1);
                    }
                    else
                    {
                        posYSt = posBgY - wpHudStackYOff;
                        tempSt.RenderModel().SetTexture(0, tex_offset_gfx_HudElements + 2);
                    }
                    HMap_.PositionHudElement(tempSt, pz_st + pz_off, posBgX, posYSt, wpHudElemWidth, wpHudElemHeight, generic_plane_rot);
                    tempSt.RenderModel().SetVisibility(0, true);
                }
                else
                {
                    tempSt.RenderModel().SetVisibility(0, false);
                }
            }


            // Position weapon element
            // -----------------------
            float posX = posBgX;
            float posY = posBgY;
            @temp = elems_weaponGFX_.actors[iSlot];
            HMap_.PositionHudElement(temp, pz_wp + pz_off, posX, posY, wpHudElemWidth * wpHudBgToWpScale, wpHudElemHeight * wpHudBgToWpScale, generic_plane_rot);
            temp.RenderModel().SetTexture(0, idxWpTex);


            // Position ammo bar element
            // -------------------------
            int ammo = Player.GetAmmo(idxWp);
            int altAmmo = Player.GetAltAmmo(idxWp);

            @tempBar = elems_weaponAmmoBar_.actors[iSlot];

            bool isAltAmmoBarSlot = false;
            if(slotAltAmmoID[iSlot] > 0)
            {
                @tempAltBar = elems_weaponAltAmmoBar_.actors[slotAltAmmoID[iSlot]-1];
                isAltAmmoBarSlot = true;
            }

            if(isAltAmmoBarSlot)
            {
                if(altAmmo > 0)
                {
                    if(altAmmo > g_maxAltAmmo[idxWp])
                    {
                        g_maxAltAmmo[idxWp] = altAmmo;
                        Log("Max alt ammo for weapon " + idxWp + " has been increased to " + altAmmo);
                    }
                    int altAmmoBarMax = g_maxAltAmmo[idxWp];
                    tempAltBar.RenderModel().SetTexture(0, tex_offset_gfx_HudBars + 3);   // red
                    tempAltBar.RenderModel().SetVisibility(0, true);

                    float pcBar = float(Clip(altAmmo, 0, altAmmoBarMax)) / float(altAmmoBarMax);
                    float barWidth = wpHudElemWidth * 0.90f * pcBar;
                    float barHeight =  wpHudElemHeight * 0.15f;
                    float posBX = posBgX - wpHudElemWidth / 2.0f + wpHudElemWidth * 0.05f + barWidth / 2.0f;
                    float posBY = posBgY - wpHudElemHeight * 0.45f;
                    if(ammo > 0) 
                    {
                        barHeight /= 2.0f;
                        posBY += barHeight / 2.0f;
                    }
                    HMap_.PositionHudElement(tempAltBar, pz_am + pz_off, posBX, posBY, barWidth, barHeight, generic_plane_rot);
                }
                else 
                {
                    tempAltBar.RenderModel().SetVisibility(0, false);
                }
            }

            if(ammo > 0)
            {
                if(ammo > g_maxAmmo[idxWp])
                {
                    g_maxAmmo[idxWp] = ammo;
                    Log("Max ammo for weapon " + idxWp + " has been increased to " + ammo);
                }
                int ammoBarMax = g_maxAmmo[idxWp];
                tempBar.RenderModel().SetTexture(0, tex_offset_gfx_HudBars + 1);   // yellow
                tempBar.RenderModel().SetVisibility(0, true);

                float pcBar = float(Clip(ammo, 0, ammoBarMax)) / float(ammoBarMax);
                float barWidth = wpHudElemWidth * 0.90f * pcBar;
                float barHeight =  wpHudElemHeight * 0.15f;
                float posBX = posBgX - wpHudElemWidth / 2.0f + wpHudElemWidth * 0.05f + barWidth / 2.0f;
                float posBY = posBgY - wpHudElemHeight * 0.45f;
                if(altAmmo > 0) 
                {
                    barHeight /= 2.0f;
                    posBY -= barHeight / 2.0f;
                }
                HMap_.PositionHudElement(tempBar, pz_am + pz_off, posBX, posBY, barWidth, barHeight, generic_plane_rot);
            }
            else
            {
                tempBar.RenderModel().SetVisibility(0, false);
            }
            


        }

        
    }
