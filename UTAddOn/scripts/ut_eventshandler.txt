// Information about the current game state
class UT_GameData
{
    int ticks;
    bool isActive;
    int level;
    bool wasHit;
    int Kills;
    int Health;
    int Shield;
    bool isInBossFight;
    int sessionID;
    
    UT_GameData()
    {
        Reset();
    }

    void Reset()
    {
        ticks = 0;
        isActive = false;
        level = 0;
        wasHit = false;
        Kills = 0;
        Health = 0;
        Shield = 0;
        isInBossFight = false;
        sessionID = 0;
    }
}

// Killstreak related data
class UT_KillStreak
{
    int killStreak_count;
    int killStreak_nKillsBegin;
    float killStreak_LastKillTime;
    bool isOnStreak;

    UT_KillStreak()
    {
        Reset();
    }
    
    void Reset()
    {
        killStreak_count = 0;
        killStreak_nKillsBegin = 0;
        killStreak_LastKillTime = 0;
        isOnStreak = false;
    }

    void Realign(int nKills, float gameTime)
    {
        Reset();
        killStreak_count = nKills;
        killStreak_LastKillTime = gameTime;
    }
}

class UT_EventsHandler
{
    kStr strCOMP = "Main";

    UT_GameData curData_;
    UT_GameData prevData_;
    UT_KillStreak streakData_;
    UT_BossFight bossData_;
    UT_AchievementFX_Stack achFXStack_;

    UT_Array_TempActor tempSpawnedHealthActorArray_;
    
    UT_Hud hud_;
    UT_SoundManager soundManager_;

    // General
    float gameTime_ = 0;
    kPuppet @pPuppet_;

    // Damage/Heal
    float damageInterval_ = 1.0;
    float last_damage_time_ = 0.0;
    float last_heal_time_ = 0.0;
    int damageInfliction_ = 0;
    float last_health_warning_time_ = -2.0;

    // Event detection
    bool spilledFirstBlood_ = false;
    bool hasKilled_ = false;
    bool gotHit_ = false;
    bool wasMapJustLoaded_ = false;
    int ownageCount_ = 0;
    


    UT_EventsHandler()
    {
        PrintVersion();
        Reset();
        prevData_.isActive = true;
    }


    void Reset()
    {

        damageInterval_ = 1.0;

        hasKilled_ = false;
        gotHit_ = false;
        damageInfliction_ = 0;
        wasMapJustLoaded_ = false;
        ownageCount_ = 0;
        last_health_warning_time_ = -2.0;

        curData_.Reset();
        prevData_.Reset();
        streakData_.Reset();
        bossData_.Reset();

        
        g_saveData.InitSaveData();

    }

    void GenerateNewSessionID()
    {
        g_saveData.sessionID = Math::RandMax(2147483647);
        Log(strCOMP, "sessionID: " + g_saveData.sessionID);
    }

    void CheckVersion()
    {
        // nothing to do
    }
    


    void DealDamage(int damage)
    {

        if(!g_gamePlaySettings.enableDamage) return;

        //Log(strCOMP, "Health: " + pPuppet_.Health() + " | Armor: " + Player.Armor());

        damageInfliction_ = damage;
        //Log(strCOMP, "Deal damage: " + damage);
        pPuppet_.CastToActor().InflictGenericDamage(pPuppet_.CastToActor(), damage);
    }


    void Heal(int health, int armor, int life = 0)
    {
        // Heal health or armor or spawn lifeforce
        // If health is maxxed, put health into armor
        // If armor is maxxed, spawn lifeforce

        if(!g_gamePlaySettings.enableHealing)
        {
            return;
        }

        int armorc = armor * 3;
        int maxHealArmor = g_saveData.maxHealArmor * 3;
        int maxHealHealth = g_saveData.maxHealHealth;

        // Heal health
        if(health > 0)
        {
            if(pPuppet_.Health() + health > maxHealHealth)
            {
                if(pPuppet_.Health() < maxHealHealth)
                {
                    // Fill up health to max and put rest into armor
                    pPuppet_.Health() = maxHealHealth;
                    Heal(0, pPuppet_.Health() + health - maxHealHealth);    // recursive
                }
                else
                {
                    // Put all health into armor
                    Heal(0, health);    // recursive
                }
            }
            else
            {
                // Put all health into health
                pPuppet_.Health() += health;
            }
        }
        
        // Heal armor
        if(armorc > 0 && g_gamePlaySettings.enableArmorHealing)
        {
            
            if(Player.Armor() + armorc > maxHealArmor)
            {
                if(Player.Armor() < maxHealArmor)
                {
                    // Fill armor to max and spawn a lifeforce
                    Player.Armor() = maxHealArmor;
                    //tempSpawnedHealthActorArray_.Push(UTAddOn::Turok::SpawnLifeForce());
                }
                else
                {
                    // Spawn a lifeforce
                    //tempSpawnedHealthActorArray_.Push(UTAddOn::Turok::SpawnLifeForce());
                }
            }
            else
            {
                // Put all health into armor
                Player.Armor() += armorc;
            }
        }

        // Spawn lifeforce
        if(life > 0)
        {
            //tempSpawnedHealthActorArray_.Push(UTAddOn::Turok::SpawnLifeForce());
        }
        
    }

    void GotDamage(int damage)
    {
        // Function is called by OnDamage and OnArmorDamage via player class

        if(damage > 0)
        {
            Log(strCOMP, "Got Damage: " + damage);
            if(damage != damageInfliction_)
            {
                damageInfliction_ = 0;
                gotHit_ = true;
            }

            if(prevData_.Shield > 0)
            {
                soundManager_.Request_PlaySound(UT_SND_EV_ARMOR_HIT);
                if(g_gamePlaySettings.enableHud)
                {
                    if(damage > 3*g_gamePlaySettings.armorDmg_Over) // no >= ! because Dmg_Over == Dmg_Norm
                    {
                        hud_.StartAnimation_ArmorHit(1);
                    }
                    else
                    {
                        hud_.StartAnimation_ArmorHit(0);
                    }
                }
            }
            else
            {
                if(g_gamePlaySettings.enableHud)
                {
                    if(damage >= g_gamePlaySettings.healthDmg_Over)
                    {
                        hud_.StartAnimation_Hit(2);
                    }
                    else if(damage >= g_gamePlaySettings.healthDmg_Norm)
                    {
                        hud_.StartAnimation_Hit(1);
                    }
                    else
                    {
                        hud_.StartAnimation_Hit(0);
                    }
                }
            }
        }

    }


    void NewGame()
    {
        kStr strHudOn, strHudOpac;
        Sys.GetCvarValue("g_drawhud", strHudOn);
        Sys.GetCvarValue("hud_opacity", strHudOpac);

        bool isTurokHudEnabled = (strHudOn.Atoi() == 1);
        bool isTurokHudTranslucent = (strHudOpac.Atof() <= 100.0f);

        if(isTurokHudEnabled && !isTurokHudTranslucent)
        {
            InfoW("Turok HUD is enabled -> UTAddOn HUD has been DISABLED!");
            InfoW("  Go to Options->GamePlay->Hud Options->Show Hud:Off to disable the Turok HUD");
            InfoW("  Alternatively, set the HUD opacity to a value smaller than 100.0");
            InfoW("  Then: Restart the game (or load a level) and the UTAddOn HUD will be enabled automatically");
            InfoW("Execute command 'call UTAddOn::Hud::Enable' to enable the UTAddOn HUD manually");
            InfoW("Note: Disabling the Turok HUD will make several in-game texts invisible! But it's recommended for the best experience");
            UTAddOn::Hud::Disable();
        }
        else 
        {
            InfoW("Turok HUD is disabled -> UTAddOn HUD has been ENABLED automatically!");
            InfoW("  Go to Options->GamePlay->Hud Options->Show Hud:On to re-enable the Turok HUD");
            InfoW("  Alternatively, set the HUD opacity to a value greater than 100.0");
            InfoW("  Then: Restart the game (or load a level) and the UTAddOn HUD will be disabled automatically");
            InfoW("Execute command 'call UTAddOn::Hud::Disable' to disable the UTAddOn HUD manually");
            InfoW("Hint: Consider disabling bobbing effects as well for the best HUD experience (Options->GamePlay->Head Bobbing Options)");
            InfoW("Note: A disabled Turok HUD will make several in-game texts invisible! But it's recommended for the best experience");
            UTAddOn::Hud::Enable();
        }

        GenerateNewSessionID();
    }


    void MapLoaded(int mapId)
    {
        // Function is called by OnBeginLevel via player class

        // Display a note to the user in the console about available console commands
        DisplayConsoleHint();
        
        CheckVersion();

        Warn(strCOMP, "Loaded map with id: " + mapId);
        Warn(strCOMP, "Level = " + UTAddOn::Turok::GetLevelId());
        

        // Get UTAddOn save variables
        g_saveData.InitSaveData();
        array<int> ksArray = g_saveData.killStreakCounter;
        g_killCounter = ksArray[0];

        // Update difficulty
        ChangeDifficultySetting(g_saveData.difficulty);

        // Set information that some map was just loaded
        wasMapJustLoaded_ = true;
        
        // Clear the hud
        hud_.Invalidate(true);

    }


    void Tick()
    {
        // Function is called by OnTick via player class

        @pPuppet_ = Player.Actor();

        prevData_ = curData_;   // Update prevData_
        UpdateGameData();       // Update curData_

        ProcessEvents();

        bool isOnOwnageStreak = (ownageCount_ > 1);
        soundManager_.Tick( isOnOwnageStreak );   // Process sound manager
        
    }


    void UpdateGameData()
    {
        curData_.ticks = PlayLoop.Ticks();
        curData_.Kills = g_killCounter;
        curData_.level = UTAddOn::Turok::GetLevelId();
        curData_.isActive = UTAddOn::Turok::IsInGame(curData_.level);
        curData_.Health = pPuppet_.Health();
        curData_.Shield = Player.Armor();
        curData_.wasHit = gotHit_;
        curData_.sessionID = g_saveData.sessionID;

        gotHit_ = false;

        gameTime_ = PlayLoop.Ticks() * GAME_DELTA_TIME;

        bool isBossLevel = UTAddOn::Turok::IsBossFight(curData_.level);
        curData_.isInBossFight = isBossLevel && UTAddOn::Turok::DetectBossFight(bossData_.pBoss);
        
    }


    void ProcessEvents()
    {
        bool isCurrentlyInGame = curData_.isActive;
        bool wasPreviouslyInGame = prevData_.isActive;

        if(!isCurrentlyInGame) 
        {
            if(wasPreviouslyInGame)
            {
                // Game ended -> Reset
                Log(strCOMP, "Inactive");
                Reset();
            }
        }
        else
        {
            if(!wasPreviouslyInGame)
            {
                // Game started
                Log(strCOMP, "Reset");
            }

            // Detect session change (savegame load detection)
            if(curData_.sessionID != prevData_.sessionID) {
                // A savegame has been loaded (probably)
                // Generate new sessionID, to detect another loading of the same savegame
                Warn(strCOMP, "SESSION ID HAS CHANGED: " + prevData_.sessionID + " -> " + curData_.sessionID);
                GenerateNewSessionID();
                Warn(strCOMP, "Generated new sessionID: " + g_saveData.sessionID);
                curData_.sessionID = g_saveData.sessionID;
            }

            // Detect level change
            bool hasLevelChanged = (curData_.level != prevData_.level) &&
                                   (curData_.level != -prevData_.level);

            if(hasLevelChanged) 
            {
                if(!wasPreviouslyInGame)
                {
                    Log(strCOMP, "New game was started");
                    if(!wasMapJustLoaded_)
                    {
                        Warn(strCOMP, "wasMapJustLoaded_ should be true, but it isnt!!");
                    }
                    NewGame();
                }
                else
                {
                    Log(strCOMP, "Level has changed");
                }
                spilledFirstBlood_ = false;
                g_saveData.firstBloodSpilled = 0;
            }

            // Prevent false events on map load
            if(wasMapJustLoaded_)
            {
                if(prevData_.Kills != curData_.Kills)
                {
                    Warn(strCOMP, "Realign prevData ...");
                    prevData_.Kills = curData_.Kills;
                    prevData_.Health = curData_.Health;
                    prevData_.Shield = curData_.Shield;
                    streakData_.Realign(curData_.Kills, gameTime_);
                    if(g_saveData.firstBloodSpilled == 1) 
                    {
                        spilledFirstBlood_ = true;
                    }
                    else
                    {
                        spilledFirstBlood_ = false;
                    }
                    ownageCount_ = 0;
                }
            }

            // Detect if kill counter has increased
            hasKilled_ = (curData_.Kills > prevData_.Kills);
            if(hasKilled_)
            {
                Log(strCOMP, "Kills: " + curData_.Kills);
            }

            // Call event processing methods
            ProcessFirstBlood();
            ProcessKillStreak();
            ProcessKillSpree();
            ProcessHealth();
            ProcessBossFight();
            
            // Update HUD
            if(g_gamePlaySettings.enableHud)
            {
                ProcessHUD();
            }
            else
            {
                hud_.Invalidate();
            }

            // At this point the map loaded information is no longer needed
            wasMapJustLoaded_ = false;
        }
    }


    void ProcessBossFight()
    {
        if(!prevData_.isInBossFight && curData_.isInBossFight)
        {
            Warn(strCOMP, "Entered Boss Fight!");
            bossData_.bossHealthItemSpawnDT = 1.0f;  // Init timer to default value
            bossData_.bossInitialHealth = bossData_.pBoss.Health();
        }

        if(!curData_.isInBossFight)
        {
            if(prevData_.isInBossFight)
            {
                Log(strCOMP, "Clear temp actor array ...");
                // Despawn everything that is left
                if(tempSpawnedHealthActorArray_.Size() > 0)
                {
                    tempSpawnedHealthActorArray_.Clear();
                }
            }
        }
        else
        {

            int bossType = bossData_.pBoss.Type();
            bool isCampaigner = (bossType == AT_AIBOSS_CAMPAINGER);
            bool isTRex = (bossType == AT_AIBOSS_TREX);
            bool isHummer = (bossType == AT_AIBOSS_HUMMER);

            bool doSpawnHealth = false;
            kStr strItem = "";

            int curBossHealth = bossData_.pBoss.Health();
            bossData_.isBossDead = (curBossHealth <= 0);
            bool didDie = (bossData_.isBossDead && bossData_.prevBossHealth > 0);
            bossData_.prevBossHealth = curBossHealth; // Update (all processing must be above)

            if(!bossData_.isBossDead)
            {
                if(curData_.Health < 20)
                {
                    strItem = "Health_Large";
                    bossData_.bossHealthItemSpawnDT = 5.0;
                    doSpawnHealth = true;
                    if(prevData_.Health >= 20)
                    {
                        // Spawn (almost) immediately
                        bossData_.last_item_spawn_time = gameTime_ - bossData_.bossHealthItemSpawnDT;
                    }
                }
                else if(curData_.Health < 40)
                {
                    strItem = "Health_Small";
                    bossData_.bossHealthItemSpawnDT = 1.0;
                    doSpawnHealth = true;
                }
                else if(curData_.Health < 60)
                {
                    strItem = "Health_Small";
                    bossData_.bossHealthItemSpawnDT = 2.0;
                    doSpawnHealth = true;
                }
                else
                {
                    doSpawnHealth = false;
                }
            }
            else if(didDie)
            {
                if(!isCampaigner && !isTRex)
                {
                    Log(strCOMP, "Boss died, spawn full health!");

                    // Spawn immediately
                    tempSpawnedHealthActorArray_.Push(UTAddOn::Turok::SpawnItemOnActor(bossData_.pBoss, "Health_Full", 300.0));
                }
                else
                {
                    Log(strCOMP, "Boss died, but DON'T spawn full health!");
                }
                doSpawnHealth = false;
            }

            if(doSpawnHealth)
            {
                float dtlis = gameTime_ - bossData_.last_item_spawn_time;
                if(dtlis > bossData_.bossHealthItemSpawnDT)
                {
                    tempSpawnedHealthActorArray_.Push(UTAddOn::Turok::SpawnItemOnActor(bossData_.pBoss, strItem, 10.0));
                    bossData_.last_item_spawn_time = gameTime_;
                    Log(strCOMP, "Spawn item '" + strItem + "' on boss");
                }
                if(dtlis < 0)
                {
                    bossData_.last_item_spawn_time = gameTime_;
                    Warn(strCOMP, "Reset bossData_.last_item_spawn_time");
                }
            }

            tempSpawnedHealthActorArray_.Update(GAME_DELTA_TIME);
        }

    }

    void ProcessFirstBlood()
    {
        if(hasKilled_ && !spilledFirstBlood_ && !wasMapJustLoaded_)
        {
            Log(strCOMP, "First blood!");
            spilledFirstBlood_ = true;
            g_saveData.firstBloodSpilled = 1;
            if(g_victimPosArray.Size() > 0)
            {
                tempSpawnedHealthActorArray_.Push(
                            UTAddOn::Turok::SpawnItemAtPosition(g_victimPosArray.At(0), 
                                                                "Health_Large", 
                                                                300.0, 
                                                                Player.Actor().SectorIndex()));
            }
            soundManager_.Request_PlaySound(UT_SND_EV_FIRST_BLOOD);
        }
    }

    void ProcessHUD()
    {
        UT_Hud_Data data;

        data.killSpree_count = g_saveData.killSpreeCounter;
        data.killSpree_remainder = g_saveData.killSpreeRemainder;

        data.killStreak_count = streakData_.killStreak_count;
        if(streakData_.isOnStreak)
        {
            float timeSinceLastKill = Clip(gameTime_ - streakData_.killStreak_LastKillTime, 0.0, g_gamePlaySettings.streakDuration);
            data.killStreak_perc = (g_gamePlaySettings.streakDuration - timeSinceLastKill) / g_gamePlaySettings.streakDuration;
        }
        else 
        {
            data.killStreak_perc = 0.0f;
        }

        data.health = pPuppet_.Health();
        data.health_heal = g_saveData.maxHealHealth;
        data.health_max = g_gamePlayParams.par_upperBoundHealth;

        data.armor = Player.Armor() / 3;
        data.armor_heal = g_saveData.maxHealArmor;
        data.armor_max = g_gamePlayParams.par_upperBoundArmor;

        data.isBossFight = curData_.isInBossFight;
        if(curData_.isInBossFight)
        {
            @data.bossActor = bossData_.pBoss;
            data.bossHealth = bossData_.pBoss.Health();
            data.bossInitialHealth = bossData_.bossInitialHealth;
        }
        else
        {
            @data.bossActor = null;
            data.bossHealth = 0;
            data.bossInitialHealth = 0;
        }

        hud_.UpdateHudData(data);
        hud_.Tick();
    }

    void ProcessHealth()
    {

        bool doDamage = false;
        bool doHeal = false;
        int nKillStreakKills = 0;

        float dtlh = gameTime_ - last_heal_time_;
        float dtld = gameTime_ - last_damage_time_;

        if(dtlh < 0 || dtld < 0)
        {
            last_heal_time_ = gameTime_;
            last_damage_time_ = gameTime_;
            Warn(strCOMP, "Reset last_damage/heal_time_");
        }

        // Armor check
        if(Player.Armor() < 0)
        {
            Player.Armor() = 0;
        }

        // Check if a health item was (supposedly) collected
        if(!wasMapJustLoaded_ )
        {
           if(curData_.Health > prevData_.Health + 5 &&
              !(prevData_.Health <= 0 && curData_.Health >= 100))
            {
                soundManager_.Request_PlaySound(UT_SND_EV_BOOSTER);
                Log(strCOMP, "Booster: Health + " + (curData_.Health - prevData_.Health));
                Log(strCOMP, "prev health: " + prevData_.Health);
                Log(strCOMP, "cur health: " + curData_.Health);
            }
            if(curData_.Shield > prevData_.Shield + 20)
            {
                soundManager_.Request_PlaySound(UT_SND_EV_SHIELD_PACK);
                Log(strCOMP, "Booster: Shield + " + (curData_.Shield - prevData_.Shield));
            }
        }

        // Determine heal and/or damage action
        if(streakData_.isOnStreak)
        {
            nKillStreakKills = curData_.Kills - streakData_.killStreak_nKillsBegin;
            
            int minStreak = 1;
            if(!g_gamePlaySettings.enableDamageInterrupt)
            {
                minStreak = 0;
            }

            if(nKillStreakKills > minStreak)
            {
                float maxHealRate = g_gamePlaySettings.maxHealRate;
                float healRate = Math::Max(1.0 / (nKillStreakKills - minStreak), maxHealRate);
                if(dtlh >= healRate)
                {
                    doHeal = true;
                    last_heal_time_ = gameTime_;
                }
            }
        }
        
        if(!streakData_.isOnStreak || !g_gamePlaySettings.enableDamageInterrupt) 
        {
            if(dtld >= damageInterval_)
            {
                doDamage = true;
                last_damage_time_ = gameTime_;
            }
        }

        // Prohibit damage after a boss has been killed
        // (player death could lead to softlock)
        if(curData_.isInBossFight && bossData_.isBossDead)
        {
            return;
        }

        // Do Heal or inflict damage (or nothing)
        if(doHeal)
        {
            Heal(1, 0);
        }
        
        if(doDamage) 
        {
            if(Player.Armor() > 0)
            {
                damageInterval_ = 2.0;
                int maxHealArmor = g_saveData.maxHealArmor;
                int damage = g_gamePlaySettings.armorDmg_Norm;
                if(Player.Armor() > maxHealArmor * 3)
                {
                    if(Player.Armor() - damage * 3 > maxHealArmor * 3)
                    {
                        damageInterval_ = 1.0;
                    }
                    damage = g_gamePlaySettings.armorDmg_Over;
                }
                DealDamage(damage * 3);
            }
            else
            {
                damageInterval_ = 1.0;
                int maxHealHealth = g_saveData.maxHealHealth;
                int damage = g_gamePlaySettings.healthDmg_60;
                if(pPuppet_.Health() > maxHealHealth)
                {
                    int tgtDamage = g_gamePlaySettings.healthDmg_Over;
                    damage = Min(pPuppet_.Health() - maxHealHealth, tgtDamage);
                    damageInterval_ = 2.0;
                }
                else if(pPuppet_.Health() > 60)
                {
                    int tgtDamage = g_gamePlaySettings.healthDmg_Norm;
                    //damage = Min(pPuppet_.Health() - 60, tgtDamage);
                    damage = tgtDamage;
                    if(pPuppet_.Health() - damage > 60)
                    {
                        damageInterval_ = 2.0;
                    }
                }
                Log(strCOMP, "Deal damage: " + damage + " with interval " + damageInterval_);
                DealDamage(damage);
            }   
        }


        // Announcements
        if(g_gamePlaySettings.enableDamage && Player.Armor() < 1)
        {
            if(last_health_warning_time_ > gameTime_)
            {
                Log(strCOMP, "Reset last_health_warning_time_");
                last_health_warning_time_ = -2.0f;
            }

            int newHealth = pPuppet_.Health();
            int oldHealth = prevData_.Health;
            if(oldHealth == 1 && newHealth > 1 && !wasMapJustLoaded_)
            {
                // Last second save
                Log(strCOMP, "oldHealth: " + oldHealth);
                Log(strCOMP, "newHealth: " + newHealth);
                soundManager_.Request_PlaySound(UT_SND_EV_LAST_SECOND_SAVE);
            }
            else if(newHealth <= 10 && oldHealth != newHealth && newHealth < oldHealth)
            {
                // 10 seconds countdown
                switch(newHealth)
                {
                    case 10:
                        soundManager_.Request_PlaySound(UT_SND_T_10);
                        break;
                    case 9:
                        soundManager_.Request_PlaySound(UT_SND_T_9);
                        break;
                    case 8:
                        soundManager_.Request_PlaySound(UT_SND_T_8);
                        break;
                    case 7:
                        soundManager_.Request_PlaySound(UT_SND_T_7);
                        break;
                    case 6:
                        soundManager_.Request_PlaySound(UT_SND_T_6);
                        break;
                    case 5:
                        soundManager_.Request_PlaySound(UT_SND_T_5);
                        break;
                    case 4:
                        soundManager_.Request_PlaySound(UT_SND_T_4);
                        break;
                    case 3:
                        soundManager_.Request_PlaySound(UT_SND_T_3);
                        break;
                    case 2:
                        soundManager_.Request_PlaySound(UT_SND_T_2);
                        break;
                    case 1:
                        soundManager_.Request_PlaySound(UT_SND_T_1);
                        break;
                    case 0:
                        soundManager_.Request_PlaySound(UT_SND_EV_DEAD);
                        break;
                }
            }
            else if(gameTime_ - last_health_warning_time_ >= 2.0f)
            {
                if(newHealth <= 20 && oldHealth > 20)
                {
                    // 20 health warning
                    last_health_warning_time_ = gameTime_;
                    soundManager_.Request_PlaySound(UT_SND_T_20);
                }
                else if(newHealth <= 30 && oldHealth > 30)
                {
                    // 30 health warning
                    last_health_warning_time_ = gameTime_;
                    soundManager_.Request_PlaySound(UT_SND_T_30);
                }
                else if(newHealth <= 60 && oldHealth > 60)
                {
                    // 60 health warning
                    last_health_warning_time_ = gameTime_;
                    soundManager_.Request_PlaySound(UT_SND_T_60);
                }
            }

        }

    }

    void ProcessKillStreak()
    {

        float timeSinceLastKill = gameTime_ - streakData_.killStreak_LastKillTime;
        if(timeSinceLastKill < 0)
        {
            float prevGameTime = prevData_.ticks * GAME_DELTA_TIME;
            Warn(strCOMP, "Reset LastKillTime from: " + streakData_.killStreak_LastKillTime);
            streakData_.killStreak_LastKillTime -= prevGameTime;
            Warn(strCOMP, "To: " + streakData_.killStreak_LastKillTime);
        }

        bool killingStreakHasEnded = (timeSinceLastKill > g_gamePlaySettings.streakDuration);

        int nKillStreakKills = curData_.Kills - streakData_.killStreak_nKillsBegin;

        if(nKillStreakKills < 0)
        {
            Warn(strCOMP, "Reset killStreak_nKillsBegin");
            streakData_.killStreak_nKillsBegin = curData_.Kills;
            nKillStreakKills = 0;
        }

        streakData_.killStreak_count = nKillStreakKills;

        if (!hasKilled_)
        {
            if(killingStreakHasEnded)
            {
                // End the current streak
                if(streakData_.isOnStreak && nKillStreakKills > 0)
                {
                    Log("Streak end");
                    if(g_gamePlaySettings.enableDamageInterrupt)
                    {
                        // Ensure damage is taken immediately after streak end
                        last_damage_time_ = 0; //gameTime_ - damageInterval_;
                    }
                }
                streakData_.isOnStreak = false;
                streakData_.killStreak_nKillsBegin = curData_.Kills;
                if(ownageCount_ > 1) Warn(strCOMP, "EndStreak: ownageCount = " + ownageCount_);
                switch(ownageCount_)
                {
                    case 0:
                    case 1:
                        // do nothing
                        break;
                    case 2:
                        Info("Impressive!");
                        soundManager_.Request_PlaySound(UT_SND_EV_IMPRESSIVE);
                        break;
                    case 3:
                        Info("Excellent!");
                        soundManager_.Request_PlaySound(UT_SND_EV_EXCELLENT);
                        break;
                    case 4:
                        Info("Termination!");
                        soundManager_.Request_PlaySound(UT_SND_EV_TERMINATION);
                        break;
                    case 5:
                        Info("Bloodbath!");
                        soundManager_.Request_PlaySound(UT_SND_EV_BLOODBATH);
                        break;
                    case 6:
                        Info("Eradication!");
                        soundManager_.Request_PlaySound(UT_SND_EV_ERADICATION);
                        break;
                    case 7:
                        Info("Massacre!");
                        soundManager_.Request_PlaySound(UT_SND_EV_MASSACRE);
                        break;
                    case 8:
                        Info("Extermination!");
                        soundManager_.Request_PlaySound(UT_SND_EV_EXTERMINATION);
                        break;
                    default:
                        Info("Unreal!!! :O");
                        soundManager_.Request_PlaySound(UT_SND_EV_UNREAL);
                }
                ownageCount_ = 0;
                soundManager_.ReduceSpreeQueueToHighest();  // Prevent playing the whole list of sprees that happend in the meantime
            }
        }  
        
        // ----
        
        if(hasKilled_)
        {

            // Enable streak
            streakData_.isOnStreak = true;

            // Update last kill time
            streakData_.killStreak_LastKillTime = gameTime_;

            // Decide wich sound to play
            if(nKillStreakKills > 0)
            {
                // Spawn FX effects on kills
                // -------------------------

                int secIdx = Player.Actor().SectorIndex();
                float playerHeight = pPuppet_.ViewHeight();
                kVec3 playerEyePos = Player.Actor().Origin() + kVec3(0,0,playerHeight);

                float nearClip = 150.0;
                float minDistClip = 25.0;
                float farClip = 1000.0;
                float farInterp = 500.0;
                float zOffNear = -50;
                float zOffFar = 10;

                kVec3 fxPos;
                while(g_victimPosArray.Pop(fxPos))
                {
                    float distInterp = Clip(playerEyePos.Distance(fxPos), nearClip, farInterp);
                    float zInterp = Math::Lerp(zOffNear, zOffFar, distInterp / (farInterp - nearClip));

                    // Determine final spawn position
                    kVec3 fxSpawnPos = fxPos + kVec3(0,0,zInterp);
                    kVec3 vDir = (fxSpawnPos - playerEyePos);
                    float distClip = Norm(vDir);
                    vDir.Normalize();
                    fxSpawnPos = playerEyePos + vDir * Clip(distClip - minDistClip, nearClip, farClip);
                    achFXStack_.Reset();
                    achFXStack_.lastStreakFXSpawnPos = fxSpawnPos;

                    int fxid = nKillStreakKills - g_victimPosArray.Size();
                    switch(fxid)
                    {
                        case 1:
                            Game.SpawnFx("UTAddOn/fx/KS_Kill.kfx", fxSpawnPos, secIdx);
                            break;
                        case 2:
                            Game.SpawnFx("UTAddOn/fx/KS_DoubleKill.kfx", fxSpawnPos, secIdx);
                            break;
                        case 3:
                            Game.SpawnFx("UTAddOn/fx/KS_MultiKill.kfx", fxSpawnPos, secIdx);
                            break;
                        case 4:
                            Game.SpawnFx("UTAddOn/fx/KS_MegaKill.kfx", fxSpawnPos, secIdx);
                            break;
                        case 5:
                            Game.SpawnFx("UTAddOn/fx/KS_UltraKill.kfx", fxSpawnPos, secIdx);
                            break;
                        case 6:
                            Game.SpawnFx("UTAddOn/fx/KS_MonsterKill.kfx", fxSpawnPos, secIdx);
                            break;
                        case 7:
                            Game.SpawnFx("UTAddOn/fx/KS_LudicrousKill.kfx", fxSpawnPos, secIdx);
                            break;
                        case 8:
                            Game.SpawnFx("UTAddOn/fx/KS_HolyShit.kfx", fxSpawnPos, secIdx);
                            break;
                        default:
                            Game.SpawnFx("UTAddOn/fx/KS_Ownage.kfx", fxSpawnPos, secIdx);
                            ownageCount_++;
                            Warn(strCOMP, "Streak: ownageCount = " + ownageCount_);
                    }
                }


                // Play sounds on kills (and trigger achievements)
                // --------------------
                switch(nKillStreakKills)
                {
                    case 1:
                        if(g_gamePlaySettings.enableDamageInterrupt)
                        {
                            soundManager_.Request_PlaySound(UT_SND_KS_AVERTED);
                        }
                        Log("Streak begin");
                        break;
                    case 2:
                        soundManager_.Request_PlaySound(UT_SND_KS_DOUBLE_KILL);
                        Log("Double Kill");
                        break;
                    case 3:
                        soundManager_.Request_PlaySound(UT_SND_KS_MULTI_KILL);
                        Log("Multi Kill");
                        break;
                    case 4:
                        soundManager_.Request_PlaySound(UT_SND_KS_MEGA_KILL);
                        Log("Mega Kill");
                        break;
                    case 5:
                        soundManager_.Request_PlaySound(UT_SND_KS_ULTRA_KILL);
                        Log("Ultra Kill");
                        break;
                    case 6:
                        soundManager_.Request_PlaySound(UT_SND_KS_MONSTER_KILL);
                        Log("Monster Kill");
                        break;
                    case 7:
                        soundManager_.Request_PlaySound(UT_SND_KS_LUDICROUS_KILL);
                        Log("Ludicrous Kill");
                        Achievement_HighKillingStreak(1);
                        break;
                    case 8:
                        soundManager_.Request_PlaySound(UT_SND_KS_HOLY_SHIT);
                        Log("Holy Shit");
                        Achievement_HighKillingStreak(3);
                        break;
                    case 9:
                        soundManager_.Request_PlaySound(UT_SND_KS_OWNAGE);
                        Log("Ownage");
                        Achievement_HighKillingStreak(5);
                        break;
                    default: // >= 10
                        soundManager_.Request_PlaySound(UT_SND_KS_OWNAGE);
                        Log("Ownage");
                        Achievement_HighKillingStreak(1);
                }

                // Update save data
                array<int> ksArray = g_saveData.killStreakCounter;
                int kscIncIdx = Clip(nKillStreakKills - 1, 0, 8);
                ksArray[kscIncIdx]++;
                ksArray[0] = g_killCounter;
                g_saveData.killStreakCounter = ksArray;

            }
        }
   
    }

    
    void ProcessKillSpree()
    {
        
        int nKillSpreeKills = curData_.Kills - g_saveData.killSpreeKillsBegin;
        bool isSpreeInterrupted = curData_.wasHit && curData_.Shield <= 0;

        if(nKillSpreeKills > 0 && isSpreeInterrupted)
        {
            // Reset spree
            g_saveData.killSpreeKillsBegin = curData_.Kills;
            g_saveData.killSpreeCounter = 0;
            g_saveData.killSpreeRemainder = 0;
            soundManager_.Request_PlaySound(UT_SND_EV_DENIED);
            Log("Killing spree was interrupted");
            return;
        }

        if(!hasKilled_)
        {
            return;
        }

        // Identify spree occurrence
        int remainder = nKillSpreeKills % 5;
        int nSpree = (nKillSpreeKills - remainder) / 5;
        g_saveData.killSpreeRemainder = remainder;

        if(nSpree > g_saveData.killSpreeCounter)
        {

            g_saveData.killSpreeCounter = nSpree;

            switch(nSpree)
            {
                case 1:
                    soundManager_.Request_PlaySound(UT_SND_KP_SPREE);
                    Log("Killing Spree");
                    break;
                case 2:
                    soundManager_.Request_PlaySound(UT_SND_KP_RAMPAGE);
                    Log("Rampage");
                    break;
                case 3:
                    soundManager_.Request_PlaySound(UT_SND_KP_DOMINATING);
                    Log("Dominating");
                    break;
                case 4:
                    soundManager_.Request_PlaySound(UT_SND_KP_UNSTOPPABLE);
                    Log("Unstoppable");
                    break;
                case 5:
                    soundManager_.Request_PlaySound(UT_SND_KP_GODLIKE);
                    Log("Godlike");
                    break;
                default: // >= 6
                    soundManager_.Request_PlaySound(UT_SND_KP_WHICKED_SICK);
                    Log("Whicked Sick");
            }

            // Each spree is an achievement
            Achievement_KillingSpree(nSpree);
            
        }


    }


    void SpawnHealAchievementFX(int amount, int type)
    {

        int secIdx = Player.Actor().SectorIndex();
        float playerViewHeight = pPuppet_.ViewHeight();

        achFXStack_.stackPos++;
        float heightAdjust = 15.0f * achFXStack_.stackPos;

        // Determine final spawn position
        kVec3 fxSpawnPos = achFXStack_.lastStreakFXSpawnPos - ez * heightAdjust;
        switch(type)
        {
            case UTA_ACHV_FX_MAX_HEALTH:
                if(amount <= 5)
                {
                    kStr strFX = "UTAddOn/fx/MaxHealHealthInc" + amount + ".kfx";
                    Game.SpawnFx(strFX, fxSpawnPos, secIdx);
                }
                else
                {
                    Warn(strCOMP, "Could not spawn FX for max heal health increased -> invalid amount: " + amount);
                }
                break;

            case UTA_ACHV_FX_MAX_ARMOR:
                if(amount == 1 || amount == 5)
                {
                    kStr strFX = "UTAddOn/fx/MaxHealArmorInc" + amount + ".kfx";
                    Game.SpawnFx(strFX, fxSpawnPos, secIdx);
                }
                else
                {
                    Warn(strCOMP, "Could not spawn FX for max heal armor increased -> invalid amount: " + amount);
                }
                break;

            case UTA_ACHV_FX_AMMO:
                if(amount > 0 && amount <= 30 && amount % 5 == 0 || amount == 100)
                {
                    if(amount == 100) 
                    {
                        Game.SpawnFx("UTAddOn/fx/AmmoGift100.kfx", fxSpawnPos, secIdx);
                    }
                    else
                    {
                        kStr strFX = "UTAddOn/fx/AmmoGift" + (amount/5) + ".kfx";
                        Game.SpawnFx(strFX, fxSpawnPos, secIdx);
                    }
                }
                else
                {
                    Warn(strCOMP, "Could not spawn FX for ammo gift -> invalid amount: " + amount);
                }
                break;

            case UTA_ACHV_FX_FULL_ARMOR:
                {
                    kStr strFX = "UTAddOn/fx/ArmorGift50.kfx";
                    Game.SpawnFx(strFX, fxSpawnPos, secIdx);
                }
                break;
        }

    }

    void Achievement_HighKillingStreak(int amount)
    {
        // Ludicrous Kill: amount = 1
        // Holy shit:      amount = 3
        // Ownage:         amount = 5
        // Ownage+:        amount = 1 (each)

        // Increase max health heal by <amount>
        // If already maxxed, increase max armor heal by 1
        // If already maxxed, do nothing

        if(!g_gamePlaySettings.enableAchievements)
        {
            return;
        }

        int maxHealHealth = g_saveData.maxHealHealth;
        int maxHealArmor = g_saveData.maxHealArmor;

        if(maxHealHealth < g_gamePlayParams.par_upperBoundHealth)
        {
            maxHealHealth = Min(maxHealHealth + amount, g_gamePlayParams.par_upperBoundHealth);
            int actualAmount = maxHealHealth - g_saveData.maxHealHealth;
            SpawnHealAchievementFX(actualAmount, UTA_ACHV_FX_MAX_HEALTH);
            Info("Updated max health to: " + maxHealHealth);
            g_saveData.maxHealHealth = maxHealHealth;
        }
        else if(maxHealArmor < g_gamePlayParams.par_upperBoundArmor)
        {
            // If max health is reached, increase armor max health
            maxHealArmor = Min(maxHealArmor + 1, g_gamePlayParams.par_upperBoundHealth);
            int actualAmount = maxHealArmor - g_saveData.maxHealArmor;
            SpawnHealAchievementFX(actualAmount, UTA_ACHV_FX_MAX_ARMOR);
            Info("Updated max armor to: " + maxHealArmor);
            g_saveData.maxHealArmor = maxHealArmor;
        }
        else
        {
            Log(strCOMP, "MaxHealthIncreased achievement wihtout effect: already maxxed");
        }
    }


    void Achievement_KillingSpree(int nSpree)
    {
        // For max spree:
        // - Give full armor
        // - Give full ammo
        // - Increase max armor by 5
        // For lower sprees:
        // - Give 5 * nSpree % ammo

        if(!g_gamePlaySettings.enableAchievements)
        {
            return;
        }

        // Give ammo for current weapon
        bool isMaxSpree = (nSpree >= 6);
        int iWP = Player.CurrentWeapon();
        UTAddOn::Turok::WeaponInfo wpInfo = UTAddOn::Turok::GetWeaponInfo(iWP);

        if(!wpInfo.ownsWeapon)
        {
            Warn(strCOMP, "Internal error: Weapon ownage is invalid");
            return;
        }

        Log(strCOMP, "current ammo: " + wpInfo.ammo);
        Log(strCOMP, "current altAmmo: " + wpInfo.altAmmo);

        if(isMaxSpree)
        {
            // Full ammo (and alt ammo, if already present)!
            Log(strCOMP, "Give Full Ammo!");
            int ammoAdd = wpInfo.maxAmmo - wpInfo.ammo;
            Player.GiveWeapon(iWP, ammoAdd);
            Log(strCOMP, "Gave ammo: " + ammoAdd);
            wpInfo = UTAddOn::Turok::GetWeaponInfo(iWP);
            Log(strCOMP, "current ammo: " + wpInfo.ammo);
            Log(strCOMP, "current altAmmo: " + wpInfo.altAmmo);
            if(wpInfo.altAmmo > 0)
            {
                int altammoAdd = wpInfo.maxAltAmmo - wpInfo.altAmmo;
                Player.ConsumeAmmo(-altammoAdd);    // Gives altAmmo, but only if altAmmo is not 0 (normal ammo otherwise)
                Log(strCOMP, "Gave alt ammo: " + altammoAdd);
                wpInfo = UTAddOn::Turok::GetWeaponInfo(iWP);
                Log(strCOMP, "current ammo: " + wpInfo.ammo);
                Log(strCOMP, "current altAmmo: " + wpInfo.altAmmo);
            }
            
            soundManager_.Request_PlaySound(UT_SND_EV_AMMO);
            SpawnHealAchievementFX(100, UTA_ACHV_FX_AMMO);
        }
        else
        {
            // Give alt ammo if still owns some and if normal ammo is full
            // Give normal ammo otherwise

            // Give 5*nSpree % of maxAmmo
            float pcAmmoInc = 5.0f * float(Clip(nSpree, 0, 6));
            bool wasAmmoAdded = false;

            // Give normal ammo
            int ammoAddDesired = int( Math::Ceil(pcAmmoInc / 100.0f * float(wpInfo.maxAmmo)) );
            int ammoAdding = Min(ammoAddDesired, wpInfo.maxAmmo - wpInfo.ammo); // Add only what fits
            int curAmmo = wpInfo.ammo;
            Player.GiveWeapon(iWP, ammoAdding);
            wpInfo = UTAddOn::Turok::GetWeaponInfo(iWP);
            int ammoActuallyGiven = wpInfo.ammo - curAmmo;

            int ammoOver;
            if(ammoActuallyGiven > 0)
            {
                Log(strCOMP, "Gave ammo: " + ammoActuallyGiven);
                wasAmmoAdded = true;

                // Put rest-ammo into alt ammo
                ammoOver = ammoAddDesired - ammoActuallyGiven;
                if(ammoOver > 0)
                {
                    // Reduce ammo amount by the maxAltAmmo/maxAmmo ratio
                    ammoOver = int(float(ammoOver) * float(wpInfo.maxAltAmmo)/float(wpInfo.maxAmmo));
                }
            }
            else
            {
                // normal ammo is already full -> put everything into alt ammo
                ammoOver = int( Math::Ceil(pcAmmoInc / 100.0f * float(wpInfo.maxAltAmmo)) );
            }

            // Give alt ammo
            if(wpInfo.altAmmo > 0 && ammoOver > 0)
            {
                int altAmmoAdd = Min(ammoOver, wpInfo.maxAltAmmo - wpInfo.altAmmo);
                Player.ConsumeAmmo(-altAmmoAdd);    // Gives altAmmo, but only if altAmmo is not 0 (normal ammo otherwise)
                Log(strCOMP, "Gave alt ammo: " + altAmmoAdd);
                wasAmmoAdded = true;
            }
        
            if(wasAmmoAdded)
            {
                SpawnHealAchievementFX(int(pcAmmoInc), UTA_ACHV_FX_AMMO);
                soundManager_.Request_PlaySound(UT_SND_EV_AMMO);
            }
            wpInfo = UTAddOn::Turok::GetWeaponInfo(iWP);
            Log(strCOMP, "current ammo: " + wpInfo.ammo);
            Log(strCOMP, "current altAmmo: " + wpInfo.altAmmo);
        }
        

        // Give full armor (50)
        // Increase max heal armor by 5
        // Reset killing spree
        if(nSpree >= 6)
        {
            // Set armor to 50 (or above, if already)
            Player.Armor() = Max(50 * 3, Player.Armor());
            hud_.StartAnimation_FullSpree();
            SpawnHealAchievementFX(0, UTA_ACHV_FX_FULL_ARMOR);

            // Increase max heal armor by 5
            int newMaxHealArmor = Min(g_saveData.maxHealArmor + 5, g_gamePlayParams.par_upperBoundArmor);
            if(newMaxHealArmor > g_saveData.maxHealArmor)
            {
                g_saveData.maxHealArmor = newMaxHealArmor;
                SpawnHealAchievementFX(5, UTA_ACHV_FX_MAX_ARMOR);
                Info("Updated max armor to: " + newMaxHealArmor);
            }

            // Reset spree count to 0
            g_saveData.killSpreeCounter = 0;
            g_saveData.killSpreeRemainder = 0;
            g_saveData.killSpreeKillsBegin = curData_.Kills;
            
        }
        

    }
