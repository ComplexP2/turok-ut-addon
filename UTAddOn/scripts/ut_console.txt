void Help()
{
    Print("UTAddOn - Available commands:");
    Print("  call UTAddOn::Help");
    Print("  call UTAddOn::Status");
    Print("  call UTAddOn::Hud::Enable");
    Print("  call UTAddOn::Hud::Disable");
    Print("  call UTAddOn::Difficulty::Off");
    Print("  call UTAddOn::Difficulty::Baby");
    Print("  call UTAddOn::Difficulty::Easy");
    Print("  call UTAddOn::Difficulty::Normal");
    Print("  call UTAddOn::Difficulty::Hard");
    Print("  call UTAddOn::Difficulty::Hardcore");
}
void Status()
{
    kStr strDifficulty = "Error";
    switch(g_saveData.difficulty)
    {
        case UTA_DIFF_OFF:
            strDifficulty = "Off";
            break;
        case UTA_DIFF_BABY:
            strDifficulty = "Baby";
            break;
        case UTA_DIFF_EASY:
            strDifficulty = "Easy";
            break;
        case UTA_DIFF_NORMAL:
            strDifficulty = "Normal";
            break;
        case UTA_DIFF_HARD:
            strDifficulty = "Hard";
            break;
        case UTA_DIFF_HARDCORE:
            strDifficulty = "Hardcore";
            break;
    }

    Print("UTAddOn - Status:");
    Print(" - Difficulty: " + strDifficulty);
    Print(" - Max healing: " + g_saveData.maxHealHealth);
    Print(" - Max armor healing: " + g_saveData.maxHealArmor);
    Print(" - Kills: " + g_saveData.killStreakCounter[0]);
    Print(" - Double Kills: " + g_saveData.killStreakCounter[1]);
    Print(" - Multi Kills: " + g_saveData.killStreakCounter[2]);
    Print(" - Mega Kills: " + g_saveData.killStreakCounter[3]);
    Print(" - Ultra Kills: " + g_saveData.killStreakCounter[4]);
    Print(" - Monster Kills: " + g_saveData.killStreakCounter[5]);
    Print(" - Ludicrous Kills: " + g_saveData.killStreakCounter[6]);
    Print(" - Holy Shit: " + g_saveData.killStreakCounter[7]);
    Print(" - Ownage: " + g_saveData.killStreakCounter[8]);
}

void EnableDamage()
{
    g_gamePlaySettings.enableDamage = true;
    Info("Damage enabled");
}
void DisableDamage()
{
    g_gamePlaySettings.enableDamage = false;
    Info("Damage disabled");
}
void EnableLog()
{
    g_enableLog = true;
    Info("Dev log enabled");
}
void DisableLog()
{
    g_enableLog = false;
    Info("Dev log disabled");
}
}   // NS: Dev

namespace Difficulty
{
void Off()
{
    // Sounds and fx only
    ChangeDifficultySetting(UTA_DIFF_OFF);
    Info("Changed difficulty to: OFF");
    Info(" - Streak duration: 2.5s");
    Info(" - No continuous damage");
    Info(" - No healing");
    Info(" - No achievements");
    Info(" -> Game behaves normally, only effects and sounds are active");
}
void Baby()
{
    ChangeDifficultySetting(UTA_DIFF_BABY);
    Info("Changed difficulty to: BABY");
    Info(" - Streak duration: 3.5s");
    Info(" - No continuous damage");
    Info(" - Armor healing");
    Info(" - Unlimited healing rate");
    Info(" - All achievements");
}
void Easy()
{
    ChangeDifficultySetting(UTA_DIFF_EASY);
    Info("Changed difficulty to: EASY");
    Info(" - Streak duration: 3.0s");
    Info(" - Continuous damage");
    Info(" - Armor healing");
    Info(" - Healing rate is limited to 10 HP/s");
    Info(" - All achievements");
}
void Normal()
{
    ChangeDifficultySetting(UTA_DIFF_NORMAL);
    Info("Changed difficulty to: NORMAL");
    Info(" - Streak duration: 2.5s");
    Info(" - Continuous damage");
    Info(" - Healing rate is limited to 5 HP/s");
    Info(" - No armor healing");
    Info(" - All achievements");
}
void Hard()
{
    ChangeDifficultySetting(UTA_DIFF_HARD);
    Info("Changed difficulty to: HARD");
    Info(" - Streak duration: 2.5s");

    Info(" - All achievements");
}
void Hardcore()
{
    ChangeDifficultySetting(UTA_DIFF_HARDCORE);
    Info("Changed difficulty to: HARDCORE");
    Info(" - Streak duration: 2.5s");
    Info(" - Continuous damage is never interrupted");

    Info(" - No achievements");
}

}   // NS: Difficulty
