/*
==============================================================
TurokInsect
==============================================================
*/

final class TurokInsect : TurokEnemy
{
    TurokInsect(kActor @actor)
    {
        super(actor);
    }
    
    /*
    ==============================================================
    OnDeath
    ==============================================================
    */
    
    void OnDeath(kActor @killer, kDictMem @damageDef)
    {
        UTAddOn::UT_Addon_Hook_EnemyKilled(killer, self);
        self.AnimSpeed() = 4.0f;
    }
}
